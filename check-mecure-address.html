Checking Mecure Clinics Address in Supabase

async function checkAddress() { const resultDiv = document.getElementById('result'); try { const { data: orgs, error } = await supabaseClient .from('organizations') .select('*') .eq('name', 'Mecure Clinics') .single(); if (error) { resultDiv.innerHTML = `
Error: ${JSON.stringify(error, null, 2)}
`; return; } resultDiv.innerHTML = `

Mecure Clinics Data in Supabase:

${JSON.stringify(orgs, null, 2)}

Address Fields:

  • Name: ${orgs.name || 'NOT SET'}
  • Address: ${orgs.address || 'NOT SET'}
  • Address Line 1: ${orgs.address_line1 || 'NOT SET'}
  • Address Line 2: ${orgs.address_line2 || 'NOT SET'}
  • City: ${orgs.city || 'NOT SET'}
  • State: ${orgs.state || 'NOT SET'}
  • Country: ${orgs.country || 'NOT SET'}
  • Phone: ${orgs.phone || 'NOT SET'}
  • Org Code: ${orgs.org_code || 'NOT SET'}
`; } catch (error) { resultDiv.innerHTML = `
Exception: ${error.message}
`; } } window.addEventListener('load', checkAddress);
