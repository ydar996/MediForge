// Get all organizations function getAllOrganizations() { return JSON.parse(localStorage.getItem("organizations") || "{}"); } // Calculate age function calculateAge(dateOfBirth) { if (!dateOfBirth) return null; const today = new Date(); const birthDate = new Date(dateOfBirth); let age = today.getFullYear() - birthDate.getFullYear(); const monthDiff = today.getMonth() - birthDate.getMonth(); if (monthDiff < 0 || (monthDiff === 0 && today.getDate() < birthDate.getDate())) { age--; } return age; } // Run diagnostics window.addEventListener('load', function() { // Set timestamp document.getElementById('last-updated').textContent = new Date().toLocaleString(); const orgs = getAllOrganizations(); const orgNames = Object.keys(orgs); let allPatients = []; let patientsWithMissingAge = []; let patientsWithMissingGender = []; let patientsWithMissingRace = []; let diagnosesWithUnknownCode = []; let totalPatients = 0; let totalDiagnoses = 0; let patientsWithDiagnoses = 0; // Collect all patient data orgNames.forEach(orgName => { const patients = JSON.parse(localStorage.getItem(`${orgName}_patients`) || "[]"); patients.forEach(patient => { totalPatients++; // Check both dateOfBirth and dob fields (some patients may have different field names) const birthDate = patient.dateOfBirth || patient.dob; const age = calculateAge(birthDate); const hasGender = patient.gender && patient.gender !== 'Unknown'; const hasRace = patient.race && String(patient.race).trim(); const diagnoses = patient.diagnoses || []; // Check for missing data - Debug log to see what's happening console.log(`Patient ${patient.id}: dateOfBirth="${patient.dateOfBirth}", dob="${patient.dob}", birthDate="${birthDate}", age=${age}`); if (!birthDate || age === null) { patientsWithMissingAge.push({ org: orgName, id: patient.id, name: `${patient.firstName} ${patient.lastName}`, dateOfBirth: birthDate || 'NOT SET', gender: patient.gender || 'NOT SET', race: patient.race || 'NOT SET' }); } if (!hasGender) { patientsWithMissingGender.push({ org: orgName, id: patient.id, name: `${patient.firstName} ${patient.lastName}`, gender: patient.gender || 'NOT SET', age: age !== null ? age : 'UNKNOWN', dateOfBirth: birthDate || 'NOT SET' }); } if (!hasRace) { patientsWithMissingRace.push({ org: orgName, id: patient.id, name: `${patient.firstName} ${patient.lastName}`, race: hasRace || 'NOT SET', age: age !== null ? age : 'UNKNOWN', dateOfBirth: birthDate || 'NOT SET' }); } // Check diagnoses if (diagnoses.length > 0) { patientsWithDiagnoses++; totalDiagnoses += diagnoses.length; diagnoses.forEach(diag => { const diagnosisText = diag.diagnosis || ''; // Check if it has proper ICD code - Updated regex to handle all ICD-11 formats // Matches: A09.9, BD50.41, MC80, XA9M74, XH6116, MA19.1, ST04, 1C10.1, etc. const icdMatch = diagnosisText.match(/^([A-Z0-9]{2,6}(?:\.[A-Z0-9]{1,3})?)\s*-?\s*(.*)$/i); // Debug logging console.log(`Diagnosis: "${diagnosisText}" | Match: ${icdMatch ? 'YES' : 'NO'} | Code: ${icdMatch ? icdMatch[1] : 'N/A'}`); if (!icdMatch) { diagnosesWithUnknownCode.push({ org: orgName, patientId: patient.id, patientName: `${patient.firstName} ${patient.lastName}`, diagnosis: diagnosisText, date: diag.date, notes: diag.notes }); } }); } // Add to all patients list allPatients.push({ org: orgName, id: patient.id, name: `${patient.firstName} ${patient.lastName}`, gender: patient.gender || 'NOT SET', dateOfBirth: birthDate || 'NOT SET', age: age !== null ? age : 'UNKNOWN', race: patient.race || 'NOT SET', diagnosesCount: diagnoses.length }); }); }); // Display summary const summaryHtml = `

Patients with Diagnoses ${patientsWithDiagnoses} OK

Patients Missing Age/DOB ${patientsWithMissingAge.length} ${patientsWithMissingAge.length > 0 ? 'ISSUE' : 'OK'}

Patients Missing Race ${patientsWithMissingRace.length} ${patientsWithMissingRace.length > 0 ? 'WARNING' : 'OK'}
Diagnoses with UNKNOWN Code ${diagnosesWithUnknownCode.length} ${diagnosesWithUnknownCode.length > 0 ? 'ISSUE' : 'OK'}
`; document.getElementById('summary').innerHTML = summaryHtml; // Display patients with missing age if (patientsWithMissingAge.length > 0) { let html = `
