Get Mecure Clinics Organization ID

const supabase = window.supabase.createClient( ((window.__SUPABASE_CONFIG__||{}).url||''), ((window.__SUPABASE_CONFIG__||{}).anonKey||'') ); async function getOrgId() { const resultDiv = document.getElementById('result'); resultDiv.innerHTML = 'Loading...'; try { const { data, error } = await supabase .from('organizations') .select('*') .ilike('name', '%Mecure%'); if (error) throw error; if (data && data.length > 0) { resultDiv.innerHTML = `

Found Organizations:

${data.map(org => `
Name: ${org.name}
ID: ${org.id}
Created: ${org.created_at}
`).join('
')} `; } else { resultDiv.innerHTML = 'No organizations found with "Mecure" in the name.'; } } catch (error) { resultDiv.innerHTML = `Error: ${error.message}`; } }
