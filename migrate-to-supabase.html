} function setStatus(step, status) { const statusEl = document.getElementById(`status-${step}`); statusEl.className = `status ${status}`; statusEl.textContent = status.charAt(0).toUpperCase() + status.slice(1); } function setProgress(step, percent) { document.getElementById(`progress-${step}`).style.width = `${percent}%`; } async function analyzeData() { setStatus('analyze', 'running'); const btn = document.getElementById('btn-analyze'); const dataType = parts.slice(1).join('_'); if (!orgData[orgName]) { orgData[orgName] = { patients: 0, appointments: 0, invoices: 0, payments: 0 }; } const data = JSON.parse(localStorage.getItem(key) || '[]'); if (dataType === 'patients') orgData[orgName].patients = data.length; if (dataType === 'appointments') orgData[orgName].appointments = data.length; if (dataType.includes('invoices')) orgData[orgName].invoices = data.length; if (dataType.includes('payments')) orgData[orgName].payments = data.length; } } analysisData = orgData; // Display results const statsContainer = document.getElementById('stats-container'); statsContainer.innerHTML = ''; let totalPatients = 0, totalAppointments = 0, totalInvoices = 0, totalPayments = 0; for (const [orgName, data] of Object.entries(orgData)) { totalPatients += data.patients; totalAppointments += data.appointments; Organizations
${totalPatients}
Patients
${totalAppointments}
Appointments
${totalInvoices}
Invoices
${totalPayments}
Payments
`; document.getElementById('analyze-results').style.display = 'block'; setStatus('analyze', 'success'); btn.textContent = ' Analysis Complete'; // Enable next step if (totalPatients > 0) { document.getElementById('btn-patients').disabled = false; } if (totalAppointments > 0) { document.getElementById('btn-appointments').disabled = false; } if (totalInvoices > 0 || totalPayments > 0) { document.getElementById('btn-billing').disabled = false; } } catch (error) { console.error('Analysis error:', error); setStatus('analyze', 'error'); btn.textContent = ' Analysis Failed'; alert('Failed to analyze data: ' + error.message); } } async function migratePatients() { setStatus('patients', 'running'); const btn = document.getElementById('btn-patients'); btn.disabled = true; btn.textContent = ' Migrating...'; log('log-patients', 'Starting patient migration...', 'info'); try { // Get all patient data from localStorage let allPatients = []; let totalPatients = 0; for (const orgName of Object.keys(analysisData)) { const patientsKey = `${orgName}_patients`; const patients = JSON.parse(localStorage.getItem(patientsKey) || '[]'); totalPatients += patients.length; log('log-patients', `Found ${patients.length} patients for ${orgName}`, 'info'); // Get organization ID from Supabase const { data: orgs, error: orgError } = await supabaseClient .from('organizations') .select('id') .eq('name', orgName) .single(); if (orgError || !orgs) { log('log-patients', ` Organization "${orgName}" not found in Supabase. Skipping...`, 'warning'); continue; } const orgId = orgs.id; // Map patients to Supabase format for (const patient of patients) { allPatients.push({ organization_id: orgId, patient_id: patient.id, first_name: patient.firstName || '', last_name: patient.lastName || '', date_of_birth: patient.dateOfBirth || null, gender: patient.gender || null, phone: patient.phone || null, email: patient.email || null, address: patient.address || null, blood_group: patient.bloodGroup || null, genotype: patient.genotype || null, allergies: patient.allergies || null, chronic_conditions: patient.chronicConditions || null, emergency_contact_name: patient.emergencyContact?.name || null, emergency_contact_phone: patient.emergencyContact?.phone || null, emergency_contact_relationship: patient.emergencyContact?.relationship || null, created_at: patient.registrationDate || new Date().toISOString(), updated_at: new Date().toISOString() }); // Insert patients in batches const batchSize = 50; let migrated = 0; for (let i = 0; i < allPatients.length; i += batchSize) { const batch = allPatients.slice(i, i + batchSize); const { data, error } = await supabaseClient .from('patients') .upsert(batch, { onConflict: 'organization_id,patient_id', ignoreDuplicates: false }); } } async function migrateAppointments() { setStatus('appointments', 'running'); const btn = document.getElementById('btn-appointments'); btn.disabled = true; btn.textContent = ' Migrating...'; log('log-appointments', 'Starting appointment migration...', 'info'); try { let allAppointments = []; for (const orgName of Object.keys(analysisData)) { const appointmentsKey = `${orgName}_appointments`; const appointments = JSON.parse(localStorage.getItem(appointmentsKey) || '[]'); log('log-appointments', `Found ${appointments.length} appointments for ${orgName}`, 'info'); // Get organization ID const { data: orgs, error: orgError } = await supabaseClient .from('organizations') .select('id') .eq('name', orgName) .single(); if (orgError || !orgs) { log('log-appointments', ` Organization "${orgName}" not found. Skipping...`, 'warning'); continue; } const orgId = orgs.id; for (const appt of appointments) { // Check if patient_id is a valid UUID format const isValidUUID = (str) => { const uuidRegex = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i; return str && uuidRegex.test(str); }; // Normalize status to match database constraint const validStatuses = ['scheduled', 'confirmed', 'completed', 'cancelled', 'no-show']; let status = (appt.status || 'scheduled').toLowerCase(); // Map common variations to valid statuses if (status === 'pending') status = 'scheduled'; if (status === 'done') status = 'completed'; if (status === 'canceled') status = 'cancelled'; if (!validStatuses.includes(status)) status = 'scheduled'; allAppointments.push({ organization_id: orgId, appointment_id: appt.id, patient_id: isValidUUID(appt.patientId) ? appt.patientId : null, patient_name: appt.patientName || null, doctor_name: appt.doctorName || null, appointment_date: appt.date || null, appointment_time: appt.time || null, duration: appt.duration || 30, reason: appt.reason || null, status: status, notes: appt.notes || null, created_at: appt.createdAt || new Date().toISOString(), updated_at: new Date().toISOString() }); } } if (allAppointments.length === 0) { log('log-appointments', 'No appointments to migrate', 'warning'); setStatus('appointments', 'success'); btn.textContent = ' No Appointments to Migrate'; return; } // Deduplicate appointments by organization_id + appointment_id const uniqueAppointments = []; const seen = new Set(); for (const appt of allAppointments) { const key = `${appt.organization_id}_${appt.appointment_id}`; if (!seen.has(key)) { seen.add(key); uniqueAppointments.push(appt); } } if (uniqueAppointments.length < allAppointments.length) { const duplicates = allAppointments.length - uniqueAppointments.length; log('log-appointments', ` Removed ${duplicates} duplicate appointments`, 'warning'); } log('log-appointments', `Migrating ${uniqueAppointments.length} appointments...`, 'info'); const batchSize = 50; let migrated = 0; for (let i = 0; i < uniqueAppointments.length; i += batchSize) { const batch = uniqueAppointments.slice(i, i + batchSize); const { data, error } = await supabaseClient .from('appointments') .upsert(batch, { onConflict: 'organization_id,appointment_id', ignoreDuplicates: false }); if (error) { log('log-appointments', ` Error: ${error.message}`, 'error'); } else { migrated += batch.length; const progress = (migrated / uniqueAppointments.length) * 100; setProgress('appointments', progress); log('log-appointments', ` Migrated ${migrated}/${uniqueAppointments.length}`, 'success'); } } setStatus('appointments', 'success'); btn.textContent = ' Appointments Migrated'; log('log-appointments', ` Successfully migrated ${migrated} appointments!`, 'success'); } catch (error) { console.error('Migration error:', error); log('log-appointments', ` Failed: ${error.message}`, 'error'); setStatus('appointments', 'error'); btn.textContent = ' Migration Failed'; } } async function migrateBilling() { setStatus('billing', 'running');
