?? Populate localStorage from Supabase

?? Problem Identified:
Your localStorage is empty (0 patients found). MediForge needs patient data in localStorage to display anything.
?? Solution:
This tool will populate your localStorage with all patient data from Supabase, including the correct DOB information.

This tool will:

  • ? Fetch all patient data from Supabase
  • ? Convert it to the format MediForge expects
  • ? Populate localStorage with complete patient data
  • ? Include all the correct DOB information
const supabase = window.supabase.createClient( ((window.__SUPABASE_CONFIG__||{}).url||''), ((window.__SUPABASE_CONFIG__||{}).anonKey||'') ); const ORG_ID = '576522cc-e769-4fb4-9487-3d150857d970'; function log(message) { const logDiv = document.getElementById('log'); const timestamp = new Date().toLocaleTimeString(); logDiv.innerHTML += `[${timestamp}] ${message}
`; logDiv.scrollTop = logDiv.scrollHeight; } // Get data key for localStorage (same as MediForge uses) function getDataKey(key) { const user = JSON.parse(localStorage.getItem("user") || "{}"); return user.org ? `${user.org}_${key}` : key; } async function populateLocalStorage() { log('?? Starting localStorage population...'); try { // Step 1: Get all patient data from Supabase log('?? Fetching all patient data from Supabase...'); const { data: supabasePatients, error } = await supabase .from('patients') .select('*') .eq('organization_id', ORG_ID); if (error) throw error; log(`? Retrieved ${supabasePatients.length} patients from Supabase`); // Step 2: Convert to localStorage format (simplified to avoid JSON issues) log('?? Converting data format...'); const localStoragePatients = supabasePatients.map(patient => { // Safely handle JSON fields let allergies = []; let conditions = []; try { if (patient.allergies && patient.allergies.trim() && patient.allergies !== 'null') { allergies = JSON.parse(patient.allergies); } } catch (e) { log(`?? Warning: Could not parse allergies for ${patient.patient_id}`); allergies = []; } try { if (patient.chronic_conditions && patient.chronic_conditions.trim() && patient.chronic_conditions !== 'null') { conditions = JSON.parse(patient.chronic_conditions); } } catch (e) { log(`?? Warning: Could not parse conditions for ${patient.patient_id}`); conditions = []; } return { id: patient.patient_id, firstName: patient.first_name || '', middleName: patient.middle_name || '', lastName: patient.last_name || '', dob: patient.date_of_birth, // This is the key field! gender: patient.gender || '', maritalStatus: patient.marital_status || '', phone: patient.phone || '', email: patient.email || '', addressLine1: patient.address || '', city: patient.city || '', state: patient.state || '', country: patient.country || '', bloodGroup: patient.blood_group || '', allergies: allergies, conditions: conditions, emergencyFirstName: patient.emergency_contact_name ? patient.emergency_contact_name.split(' ')[0] : '', emergencyLastName: patient.emergency_contact_name ? patient.emergency_contact_name.split(' ').slice(1).join(' ') : '', emergencyPhone: patient.emergency_contact_phone || '', emergencyRelationship: patient.emergency_contact_relationship || '', createdDate: patient.created_at, updatedDate: patient.updated_at, // Add default fields that might be expected visits: [], immunizations: [], prescriptions: [] }; }); log(`? Converted ${localStoragePatients.length} patients to localStorage format`); // Step 3: Save to localStorage log('?? Saving to localStorage...'); const dataKey = getDataKey("patients"); localStorage.setItem(dataKey, JSON.stringify(localStoragePatients)); log(`? Saved ${localStoragePatients.length} patients to localStorage with key: ${dataKey}`); // Step 4: Verify the save const savedPatients = JSON.parse(localStorage.getItem(dataKey) || "[]"); const patientsWithDOB = savedPatients.filter(p => p.dob); log(`?? Verification: ${savedPatients.length} patients saved, ${patientsWithDOB.length} with DOB`); // Show sample of patients with DOB if (patientsWithDOB.length > 0) { log('?? Sample patients with DOB:'); patientsWithDOB.slice(0, 5).forEach(patient => { log(` ? ${patient.firstName} ${patient.lastName} (${patient.id}): ${patient.dob}`); }); } log('?? localStorage population complete!'); log('?? MediForge should now display all patients with correct DOB data'); } catch (error) { log(`? Error populating localStorage: ${error.message}`); } } function testMediForgeApp() { log('?? Testing MediForge data...'); const dataKey = getDataKey("patients"); const patients = JSON.parse(localStorage.getItem(dataKey) || "[]"); log(`?? Current localStorage data:`); log(` Total patients: ${patients.length}`); const patientsWithDOB = patients.filter(p => p.dob); log(` Patients with DOB: ${patientsWithDOB.length}`); if (patients.length > 0) { log('? Patient data is available in localStorage'); if (patientsWithDOB.length > 0) { log('? DOB data is available'); log('?? MediForge should now display patients with correct dates of birth'); } else { log('?? No DOB data found'); } log('?? Please refresh MediForge to see the changes'); } else { log('? No patient data found in localStorage'); log('?? Please run the populate tool first'); } } function clearLocalStorage() { log('??? Clearing localStorage...'); const dataKey = getDataKey("patients"); localStorage.removeItem(dataKey); log('? localStorage cleared'); log('?? You can now run the populate tool to add fresh data'); } // Initialize window.addEventListener('load', () => { log('?? localStorage Population Tool ready'); log('?? Click "Populate localStorage" to start'); });
