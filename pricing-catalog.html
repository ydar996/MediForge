window.addEventListener('load', function() { loadServices(); }); function loadServices() { const services = getPricingCatalog(); displayServices(services); } function displayServices(services) { const tbody = document.getElementById('services-list'); tbody.innerHTML = ''; services.forEach(service => { const row = tbody.insertRow(); row.innerHTML = ` ${service.code} ${service.name} ${service.category} ${formatCurrency(service.price, service.currency)} ${service.active ? 'Active' : 'Inactive'} `; }); } function filterServices() { const query = document.getElementById('search').value.toLowerCase(); const services = getPricingCatalog().filter(s => s.name.toLowerCase().includes(query) || s.code.toLowerCase().includes(query) || s.category.toLowerCase().includes(query) ); displayServices(services); } function showAddServiceForm() { document.getElementById('add-form').style.display = 'block'; document.getElementById('service-id').value = ''; document.getElementById('service-code').value = ''; document.getElementById('service-name').value = ''; document.getElementById('service-price').value = ''; document.getElementById('service-description').value = ''; document.getElementById('service-taxable').checked = true; document.getElementById('service-active').checked = true; } function editService(serviceId) { const service = getServiceById(serviceId); if (!service) return; document.getElementById('add-form').style.display = 'block'; document.getElementById('service-id').value = service.id; document.getElementById('service-code').value = service.code; document.getElementById('service-name').value = service.name; document.getElementById('service-category').value = service.category; document.getElementById('service-price').value = service.price; document.getElementById('service-description').value = service.description; document.getElementById('service-taxable').checked = service.taxable; document.getElementById('service-active').checked = service.active; } function saveService() { const serviceId = document.getElementById('service-id').value; const serviceData = { code: document.getElementById('service-code').value, name: document.getElementById('service-name').value, category: document.getElementById('service-category').value, price: parseFloat(document.getElementById('service-price').value), description: document.getElementById('service-description').value, taxable: document.getElementById('service-taxable').checked, active: document.getElementById('service-active').checked }; if (!serviceData.name || !serviceData.price) { alert('Please fill in required fields'); return; } if (serviceId) { updateServiceInCatalog(serviceId, serviceData); alert('Service updated successfully'); } else { addServiceToCatalog(serviceData); alert('Service added successfully'); } cancelEdit(); loadServices(); } function toggleServiceStatus(serviceId) { const service = getServiceById(serviceId); if (!service) return; updateServiceInCatalog(serviceId, { active: !service.active }); loadServices(); } function cancelEdit() { document.getElementById('add-form').style.display = 'none'; }
