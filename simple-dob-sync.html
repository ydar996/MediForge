?? Simple DOB Sync

?? Focused Solution:
This tool will only update the DOB data in localStorage, avoiding any JSON parsing issues.

This tool will:

  • ? Get current patients from localStorage
  • ? Get fresh DOB data from Supabase
  • ? Update only the DOB fields in localStorage
  • ? Keep all other data intact
const supabase = window.supabase.createClient( ((window.__SUPABASE_CONFIG__||{}).url||''), ((window.__SUPABASE_CONFIG__||{}).anonKey||'') ); const ORG_ID = '576522cc-e769-4fb4-9487-3d150857d970'; function log(message) { const logDiv = document.getElementById('log'); const timestamp = new Date().toLocaleTimeString(); logDiv.innerHTML += `[${timestamp}] ${message}
`; logDiv.scrollTop = logDiv.scrollHeight; } // Get data key for localStorage (same as EHR app uses) function getDataKey(key) { const user = JSON.parse(localStorage.getItem("user") || "{}"); return user.org ? `${user.org}_${key}` : key; } async function syncDOBOnly() { log('?? Starting DOB-only sync...'); try { // Step 1: Get current localStorage patients const dataKey = getDataKey("patients"); const currentPatients = JSON.parse(localStorage.getItem(dataKey) || "[]"); log(`?? Found ${currentPatients.length} patients in localStorage`); // Step 2: Get fresh DOB data from Supabase log('?? Fetching DOB data from Supabase...'); const { data: supabasePatients, error } = await supabase .from('patients') .select('patient_id, date_of_birth') .eq('organization_id', ORG_ID); if (error) throw error; log(`? Retrieved DOB data for ${supabasePatients.length} patients from Supabase`); // Step 3: Create a map of patient_id to DOB const dobMap = {}; supabasePatients.forEach(patient => { dobMap[patient.patient_id] = patient.date_of_birth; }); // Step 4: Update localStorage patients with DOB data log('?? Updating DOB data in localStorage...'); let updatedCount = 0; currentPatients.forEach(patient => { const newDOB = dobMap[patient.id]; if (newDOB && newDOB !== patient.dob) { const oldDOB = patient.dob; patient.dob = newDOB; updatedCount++; log(`? Updated ${patient.firstName} ${patient.lastName} (${patient.id}): ${oldDOB || 'NULL'} ? ${newDOB}`); } }); // Step 5: Save updated patients back to localStorage localStorage.setItem(dataKey, JSON.stringify(currentPatients)); log(`?? Saved updated patients to localStorage`); // Step 6: Verify the update const patientsWithDOB = currentPatients.filter(p => p.dob); log(`?? Final result: ${patientsWithDOB.length}/${currentPatients.length} patients have DOB data`); if (updatedCount > 0) { log(`?? Successfully updated ${updatedCount} patients with DOB data!`); log('?? Your EHR app should now show the correct dates of birth'); } else { log('?? No patients needed DOB updates'); } } catch (error) { log(`? Error syncing DOB data: ${error.message}`); } } function testEHRApp() { log('?? Testing EHR app data...'); const dataKey = getDataKey("patients"); const patients = JSON.parse(localStorage.getItem(dataKey) || "[]"); log(`?? Current localStorage data:`); log(` Total patients: ${patients.length}`); const patientsWithDOB = patients.filter(p => p.dob); log(` Patients with DOB: ${patientsWithDOB.length}`); if (patientsWithDOB.length > 0) { log('? DOB data is available in localStorage'); log('?? Patients with DOB:'); patientsWithDOB.forEach(patient => { log(` ? ${patient.firstName} ${patient.lastName} (${patient.id}): ${patient.dob}`); }); log('?? Your EHR app should now display the correct dates of birth'); log('?? Please refresh your EHR app to see the changes'); } else { log('? No DOB data found in localStorage'); log('?? Please run the DOB sync first'); } } // Initialize window.addEventListener('load', () => { log('?? Simple DOB Sync Tool ready'); log('?? Click "Sync DOB Data Only" to start'); });
