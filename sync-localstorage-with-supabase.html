?? Sync localStorage with Supabase

?? Problem Identified:
Your EHR app is loading patient data from localStorage instead of Supabase.
The DOB data is correctly stored in Supabase, but your app isn't showing it because it's using cached localStorage data.

This tool will:

  • ? Fetch fresh patient data from Supabase
  • ? Update your localStorage with the correct DOB data
  • ? Make your EHR app display the updated information
const supabase = window.supabase.createClient( ((window.__SUPABASE_CONFIG__||{}).url||''), ((window.__SUPABASE_CONFIG__||{}).anonKey||'') ); const ORG_ID = '576522cc-e769-4fb4-9487-3d150857d970'; function log(message) { const logDiv = document.getElementById('log'); const timestamp = new Date().toLocaleTimeString(); logDiv.innerHTML += `[${timestamp}] ${message}
`; logDiv.scrollTop = logDiv.scrollHeight; } // Get data key for localStorage (same as EHR app uses) function getDataKey(key) { const user = JSON.parse(localStorage.getItem("user") || "{}"); return user.org ? `${user.org}_${key}` : key; } async function syncData() { log('?? Starting localStorage sync with Supabase...'); try { // Step 1: Get fresh data from Supabase log('?? Fetching fresh data from Supabase...'); const { data: supabasePatients, error } = await supabase .from('patients') .select('*') .eq('organization_id', ORG_ID); if (error) throw error; log(`? Retrieved ${supabasePatients.length} patients from Supabase`); // Step 2: Convert Supabase format to localStorage format log('?? Converting data format...'); const localStoragePatients = supabasePatients.map(patient => { // Safely parse JSON fields let allergies = []; let conditions = []; try { if (patient.allergies && patient.allergies.trim()) { allergies = JSON.parse(patient.allergies); } } catch (e) { log(`?? Warning: Could not parse allergies for ${patient.patient_id}: ${e.message}`); allergies = []; } try { if (patient.chronic_conditions && patient.chronic_conditions.trim()) { conditions = JSON.parse(patient.chronic_conditions); } } catch (e) { log(`?? Warning: Could not parse conditions for ${patient.patient_id}: ${e.message}`); conditions = []; } return { id: patient.patient_id, firstName: patient.first_name, middleName: patient.middle_name, lastName: patient.last_name, dob: patient.date_of_birth, // This is the key fix! gender: patient.gender, maritalStatus: patient.marital_status, phone: patient.phone, email: patient.email, addressLine1: patient.address, city: patient.city, state: patient.state, country: patient.country, bloodGroup: patient.blood_group, allergies: allergies, conditions: conditions, emergencyFirstName: patient.emergency_contact_name ? patient.emergency_contact_name.split(' ')[0] : '', emergencyLastName: patient.emergency_contact_name ? patient.emergency_contact_name.split(' ').slice(1).join(' ') : '', emergencyPhone: patient.emergency_contact_phone, emergencyRelationship: patient.emergency_contact_relationship, createdDate: patient.created_at, updatedDate: patient.updated_at }; }); log(`? Converted ${localStoragePatients.length} patients to localStorage format`); // Step 3: Update localStorage log('?? Updating localStorage...'); const dataKey = getDataKey("patients"); localStorage.setItem(dataKey, JSON.stringify(localStoragePatients)); log(`? Updated localStorage with key: ${dataKey}`); // Step 4: Verify the update log('?? Verifying localStorage update...'); const storedPatients = JSON.parse(localStorage.getItem(dataKey) || "[]"); const patientsWithDOB = storedPatients.filter(p => p.dob); log(`? Verification: ${storedPatients.length} patients in localStorage, ${patientsWithDOB.length} with DOB`); // Show sample data if (patientsWithDOB.length > 0) { log('?? Sample updated patients:'); patientsWithDOB.slice(0, 3).forEach(patient => { log(` ? ${patient.firstName} ${patient.lastName} (${patient.id}): ${patient.dob}`); }); } log('?? localStorage sync complete! Your EHR app should now show the correct DOB data.'); } catch (error) { log(`? Error syncing data: ${error.message}`); } } function testEHRApp() { log('?? Current localStorage data:`); log(` Total patients: ${patients.length}`); const patientsWithDOB = patients.filter(p => p.dob); log(` Patients with DOB: ${patientsWithDOB.length}`); if (patientsWithDOB.length > 0) { log('? DOB data is available in localStorage'); log('?? Your EHR app should now display the correct dates of birth'); log('?? Please refresh your EHR app to see the changes'); } else { log('?
