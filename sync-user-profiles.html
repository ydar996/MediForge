Check and sync user profiles between Supabase Auth and users table

Step 2: View Data Summary

Step 3: Create Missing Profiles

This will create user profiles in the users table for any auth accounts that don't have one yet.

let authUsersData = []; let profilesData = []; function log(id, message, type = 'info') { const el = document.getElementById(id); el.style.display = 'block'; const entry = document.createElement('div'); entry.className = type; const timestamp = new Date().toLocaleTimeString(); entry.textContent = `[${timestamp}] ${message}`; el.appendChild(entry); el.scrollTop = el.scrollHeight; } async function checkAuthUsers() { document.getElementById('check-log').innerHTML = ''; log('check-log', 'Checking Supabase Auth users...', 'info'); log('check-log', 'Note: This requires admin access to view all auth users', 'warning'); log('check-log', 'We can only see the currently logged-in user from the browser', 'info'); // Get current session const { data: { session } } = await supabaseClient.auth.getSession(); if (session) { log('check-log', ` Current user: ${session.user.email}`, 'success'); log('check-log', ` Auth ID: ${session.user.id}`, 'info'); authUsersData = [session.user]; } else { log('check-log', 'No user logged in', 'warning'); } } async function checkUserProfiles() { document.getElementById('check-log').innerHTML = ''; log('check-log', 'Checking user profiles in users table...', 'info'); try { const { data, error, count } = await supabaseClient .from('users') .select('*', { count: 'exact' }); if (error) { log('check-log', ` Error: ${error.message}`, 'error'); } else { profilesData = data || []; log('check-log', ` Found ${count} user profiles in users table`, count > 0 ? 'success' : 'warning'); if (count > 0) { log('check-log', '\nUser Profiles:', 'info'); data.forEach(u => { log('check-log', ` - ${u.username} (${u.role}) - Auth ID: ${u.auth_user_id}`, 'info'); }); } else { log('check-log', '\nNo user profiles found!', 'warning'); log('check-log', 'This means auth accounts exist but no profiles in the users table', 'warning'); } updateSummary(); } } catch (error) { log('check-log', ` Exception: ${error.message}`, 'error'); } } function updateSummary() { const summary = document.getElementById('summary'); summary.innerHTML = `
Metric Count Status
Auth Users (visible) ${authUsersData.length} ${authUsersData.length > 0 ? 'Yes' : 'No'}
User Profiles (users table) ${profilesData.length} ${profilesData.length > 0 ? 'Yes' : 'Missing'}
`; if (profilesData.length === 0) { document.getElementById('sync-btn').disabled = false; } } async function syncProfiles() { document.getElementById('sync-log').innerHTML = ''; log('sync-log', 'Syncing user profiles...', 'info'); log('sync-log', '\nNOTE: This is a demonstration', 'warning'); log('sync-log', 'To properly sync all users, you would need:', 'info'); log('sync-log', '1. Admin access to list all auth users', 'info'); log('sync-log', '2. Or run this on the backend with service role', 'info'); log('sync-log', '\nFor now, you can manually check and create user profiles in Supabase', 'info'); // Show SQL example log('sync-log', '\nExample SQL to check auth users:', 'info'); log('sync-log', 'SELECT id, email, created_at FROM auth.users;', 'info'); }
