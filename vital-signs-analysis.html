let patientData = null; let charts = {}; // Get patient data from URL parameters function getPatientData() { const urlParams = new URLSearchParams(window.location.search); const patientId = urlParams.get('id'); if (!patientId) { document.getElementById('patient-name').textContent = 'No patient selected'; return null; } // Supabase-first: Load patients from Supabase, fallback to localStorage let patients = []; try { if (typeof window.loadPatientsWithSupabasePriority === 'function') { patients = await window.loadPatientsWithSupabasePriority(); } else { patients = JSON.parse(localStorage.getItem(getDataKey("patients")) || "[]"); } } catch (error) { patients = JSON.parse(localStorage.getItem(getDataKey("patients")) || "[]"); } const patient = patients.find(p => p.id === patientId); if (patient) { document.getElementById('patient-name').innerHTML = `${patient.firstName} ${patient.lastName} (ID: ${patient.id}) - Vital Signs Analysis`; return patient; } else { document.getElementById('patient-name').textContent = 'Patient not found'; return null; } } // Extract all vital signs from clinical notes function extractVitalSigns(patient) { const vitalSigns = []; console.log('Patient visits:', patient.visits); if (patient.visits) { patient.visits.forEach((visit, visitIndex) => { console.log(`Visit ${visitIndex}:`, visit); if (visit.soap && visit.soap.objective && visit.soap.objective.vitals) { console.log(`Visit ${visitIndex} vitals:`, visit.soap.objective.vitals); visit.soap.objective.vitals.forEach((vital, vitalIndex) => { console.log(`Vital ${vitalIndex}:`, vital); vitalSigns.push({ ...vital, visitDate: visit.date }); }); } else { console.log(`Visit ${visitIndex} has no vitals data:`, visit.soap); } }); } console.log('Final vital signs array:', vitalSigns); return vitalSigns.sort((a, b) => new Date(a.visitDate) - new Date(b.visitDate)); } // Calculate statistics for a vital sign function calculateStats(values) { if (values.length === 0) return null; const numericValues = values.map(v => parseFloat(v)).filter(v => !isNaN(v)); if (numericValues.length === 0) return null; numericValues.sort((a, b) => a - b); const sum = numericValues.reduce((a, b) => a + b, 0); const mean = sum / numericValues.length; const median = numericValues.length % 2 === 0 ? (numericValues[numericValues.length / 2 - 1] + numericValues[numericValues.length / 2]) / 2 : numericValues[Math.floor(numericValues.length / 2)]; return { count: numericValues.length, min: numericValues[0], max: numericValues[numericValues.length - 1], mean: Math.round(mean * 10) / 10, median: Math.round(median * 10) / 10, range: numericValues[numericValues.length - 1] - numericValues[0] }; } // Create chart for a vital sign function createChart(canvasId, label, data, unit) { const ctx = document.getElementById(canvasId).getContext('2d'); return new Chart(ctx, { type: 'line', data: { labels: data.map(d => new Date(d.visitDate).toLocaleDateString()), datasets: [{ label: label, data: data.map(d => d.value), borderColor: '#667eea', backgroundColor: 'rgba(102, 126, 234, 0.1)', borderWidth: 2, fill: true, tension: 0.4 }] }, options: { responsive: true, scales: { y: { beginAtZero: false, title: { display: true, text: unit } }, x: { title: { display: true, text: 'Visit Date' } } }, plugins: { legend: { display: true } } } }); } // Generate clinical insights function generateInsights(vitalSigns) { const insights = []; // Blood Pressure Analysis const bpReadings = vitalSigns.filter(v => v.systolic && v.diastolic); if (bpReadings.length > 0) { const avgSystolic = bpReadings.reduce((sum, v) => sum + parseFloat(v.systolic), 0) / bpReadings.length; const avgDiastolic = bpReadings.reduce((sum, v) => sum + parseFloat(v.diastolic), 0) / bpReadings.length; if (avgSystolic >= 140 || avgDiastolic >= 90) { insights.push({ type: 'danger', text: `Elevated blood pressure detected (Avg: ${Math.round(avgSystolic)}/${Math.round(avgDiastolic)} mmHg). Consider hypertension management.` }); } else if (avgSystolic >= 130 || avgDiastolic >= 80) { insights.push({ type: 'warning', text: `Blood pressure in Stage 1 hypertension range (Avg: ${Math.round(avgSystolic)}/${Math.round(avgDiastolic)} mmHg). Monitor closely.` }); } else { insights.push({ type: 'success', text: `Blood pressure within normal range (Avg: ${Math.round(avgSystolic)}/${Math.round(avgDiastolic)} mmHg).` }); } } // Heart Rate Analysis const hrReadings = vitalSigns.filter(v => v.hr).map(v => parseFloat(v.hr)); if (hrReadings.length > 0) { const avgHR = hrReadings.reduce((sum, v) => sum + v, 0) / hrReadings.length; if (avgHR > 100) { insights.push({ type: 'warning', text: `Elevated heart rate detected (Avg: ${Math.round(avgHR)} bpm). Consider cardiovascular assessment.` }); } else if (avgHR < 60) { insights.push({ type: 'warning', text: `Low heart rate detected (Avg: ${Math.round(avgHR)} bpm). Monitor for bradycardia.` }); } else { insights.push({ type: 'success', text: `Heart rate within normal range (Avg: ${Math.round(avgHR)} bpm).` }); } } // Temperature Analysis const tempReadings = vitalSigns.filter(v => v.temp).map(v => parseFloat(v.temp)); if (tempReadings.length > 0) { const avgTemp = tempReadings.reduce((sum, v) => sum + v, 0) / tempReadings.length; if (avgTemp > 37.5) { insights.push({ type: 'warning', text: `Elevated temperature detected (Avg: ${avgTemp.toFixed(1)}?C). Consider fever evaluation.` }); } else if (avgTemp < 36.0) { insights.push({ type: 'warning', text: `Low temperature detected (Avg: ${avgTemp.toFixed(1)}?C). Monitor for hypothermia.` }); } else { insights.push({ type: 'success', text: `Temperature within normal range (Avg: ${avgTemp.toFixed(1)}?C).` }); } } // Respiratory Rate Analysis const rrReadings = vitalSigns.filter(v => v.rr).map(v => parseFloat(v.rr)); if (rrReadings.length > 0) { const avgRR = rrReadings.reduce((sum, v) => sum + v, 0) / rrReadings.length; if (avgRR > 20) { insights.push({ type: 'warning', text: `Elevated respiratory rate (Avg: ${Math.round(avgRR)} breaths/min). Consider respiratory assessment.` }); } else if (avgRR < 12) { insights.push({ type: 'warning', text: `Low respiratory rate (Avg: ${Math.round(avgRR)} breaths/min). Monitor for respiratory depression.` }); } else { insights.push({ type: 'success', text: `Respiratory rate within normal range (Avg: ${Math.round(avgRR)} breaths/min).` }); } } // Oxygen Saturation Analysis const spo2Readings = vitalSigns.filter(v => v.o2sat).map(v => parseFloat(v.o2sat)); if (spo2Readings.length > 0) { const avgSpo2 = spo2Readings.reduce((sum, v) => sum + v, 0) / spo2Readings.length; if (avgSpo2 < 95) { insights.push({ type: 'danger', text: `Low oxygen saturation detected (Avg: ${Math.round(avgSpo2)}%). Consider oxygen therapy.` }); } else if (avgSpo2 < 97) { insights.push({ type: 'warning', text: `Borderline oxygen saturation (Avg: ${Math.round(avgSpo2)}%). Monitor closely.` }); } else { insights.push({ type: 'success', text: `Oxygen saturation within normal range (Avg: ${Math.round(avgSpo2)}%).` }); } } return insights; } // Load and display analysis function loadAnalysis() { patientData = getPatientData(); if (!patientData) return; console.log('Patient data:', patientData); const vitalSigns = extractVitalSigns(patientData); console.log('Extracted vital signs:', vitalSigns); if (vitalSigns.length === 0) { document.getElementById('stats-grid').innerHTML = '

'; return; } // Display statistics displayStatistics(vitalSigns); // Create charts createCharts(vitalSigns); // Generate insights displayInsights(vitalSigns); } // Display statistics function displayStatistics(vitalSigns) { const statsGrid = document.getElementById('stats-grid'); // Calculate total readings const totalReadings = vitalSigns.length; const dateRange = vitalSigns.length > 0 ? `${new Date(vitalSigns[0].visitDate).toLocaleDateString()} - ${new Date(vitalSigns[vitalSigns.length - 1].visitDate).toLocaleDateString()}` : 'N/A'; statsGrid.innerHTML = `
